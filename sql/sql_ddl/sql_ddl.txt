CREATE TABLE Dates (
	dateID INT,
	day INT NOT NULL,
	month INT NOT NULL,
	year INT NOT NULL,
	PRIMARY KEY (dateID) );

CREATE TABLE Stocks (
	dateID INT,
	ccl DECIMAL(19,2) NOT NULL,
	gwph DECIMAL(19,2) NOT NULL,
	jcp DECIMAL(19,2) NOT NULL,
	ibm DECIMAL(19,2) NOT NULL,
	pg DECIMAL(19,2) NOT NULL,
	xom DECIMAL(19,2) NOT NULL,
	bb DECIMAL(19,2) NOT NULL,
	fmcc DECIMAL(19,2) NOT NULL,
	axl DECIMAL(19,2) NOT NULL,
	ge DECIMAL(19,2) NOT NULL,
	bjri DECIMAL(19,2) NOT NULL,
	shldq DECIMAL(19,2) NOT NULL,
	chk DECIMAL(19,2) NOT NULL,
	aapl DECIMAL(19,2) NOT NULL,
	amzn DECIMAL(19,2) NOT NULL,
	FOREIGN KEY (dateID) REFERENCES Date (dateID),
	PRIMARY KEY (dateID) );

CREATE TABLE Commodities (
	dateID INT,
	oil DECIMAL(19,2) NOT NULL,
	gold DECIMAL(19,2) NOT NULL,
	pokemon DECIMAL(19,2) NOT NULL,
	wagner DECIMAL(19,2) NOT NULL,
	romeo DECIMAL(19,2) NOT NULL,
	valentino DECIMAL(19,2) NOT NULL,
	mario DECIMAL(19,2) NOT NULL,
	FOREIGN KEY (dateID) REFERENCES Date (dateID),
	PRIMARY KEY (dateID) );

CREATE TABLE Inflation (
	year INT,
	amount DECIMAL(19,2) NOT NULL,
	PRIMARY KEY (year) );

CREATE TABLE History (
	date INT,
	history VARCHAR(1000) NOT NULL,
	FOREIGN KEY (date) REFERENCES Date (date),
	PRIMARY KEY (date) );

CREATE TABLE Indexes (
	date INT,
	NASDAQ DECIMAL(19,2) NOT NULL,
	DOW DECIMAL(19,2) NOT NULL,
	SP DECIMAL(19,2) NOT NULL,
	bitcoin DECIMAL(19,2) NOT NULL,
	FOREIGN KEY (date) REFERENCES Date (date),
	PRIMARY KEY (date) );

CREATE TABLE Descriptions (
	itemID VARCHAR(10) NOT NULL,
	desr VARCHAR(5000),
	PRIMARY KEY (itemID)
)
